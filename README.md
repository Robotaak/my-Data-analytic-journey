        "import pandas as pd\n",
        "import matplotlib.pyplot as plt\n",
        "from google.colab import files\n",
        "\n",
        "def clean_csv(filepath):\n",
        "    print(f\"📂 Loading: {filepath}\")\n",
        "    df = pd.read_csv(filepath)\n",
        "\n",
        "    # ── 1. Fix column names ──────────────────────────\n",
        "    df.columns = df.columns.str.strip().str.lower().str.replace(' ', '_')\n",
        "    print(f\"✅ Columns cleaned: {df.columns.tolist()}\")\n",
        "\n",
        "    # ── 2. Remove duplicates ─────────────────────────\n",
        "    before = len(df)\n",
        "    df.drop_duplicates(inplace=True)\n",
        "    print(f\"✅ Duplicates removed: {before - len(df)}\")\n",
        "\n",
        "    # ── 3. Auto-detect & fix money/number columns ────\n",
        "    for col in df.columns:\n",
        "        if df[col].dtype == object:\n",
        "            cleaned = df[col].astype(str).str.replace(r'[$,\\s]', '', regex=True)\n",
        "            converted = pd.to_numeric(cleaned, errors='coerce')\n",
        "            # Only convert if MORE than 70% of values became numbers\n",
        "            if converted.notna().mean() > 0.7:\n",
        "                df[col] = converted\n",
        "                print(f\"✅ Converted to numeric: {col}\")\n",
        "\n",
        "    # ── 4. Fill remaining nulls ───────────────────────\n",
        "    null_count = df.isnull().sum().sum()\n",
        "    df.fillna(0, inplace=True)\n",
        "    print(f\"✅ Nulls filled: {null_count}\")\n",
        "\n",
        "    # ── 5. Summary report ────────────────────────────\n",
        "    print(\"\\n\" + \"=\"*45)\n",
        "    print(\"📊 CLEANING SUMMARY REPORT\")\n",
        "    print(\"=\"*45)\n",
        "    print(f\"  Rows:     {len(df)}\")\n",
        "    print(f\"  Columns:  {len(df.columns)}\")\n",
        "    print(f\"  Numerics: {df.select_dtypes(include='number').columns.tolist()}\")\n",
        "    print(f\"  Text:     {df.select_dtypes(include='object').columns.tolist()}\")\n",
        "    print(\"=\"*45)\n",
        "\n",
        "    # ── 6. Save clean file ───────────────────────────\n",
        "    out = filepath.replace('.csv', '_CLEAN.csv')\n",
        "    df.to_csv(out, index=False)\n",
        "    print(f\"\\n✅ Clean file saved: {out}\")\n",
        "\n",
        "    return df"
